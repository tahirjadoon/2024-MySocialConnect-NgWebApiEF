or running the query with group_concact

SELECT t.UserId, t.UserName, GROUP_CONCAT(t.RoleName, ', ') AS Roles
FROM (
    SELECT ur.UserId, u.UserName, ur.RoleId, r.Name AS RoleName
    FROM AspNetUserRoles ur
    INNER JOIN AspNetUsers u ON u.Id = ur.UserId
    INNER JOIN AspNetRoles r ON r.Id = ur.RoleId
    ORDER BY ur.UserId
) t
GROUP BY t.UserId, t.UserName;
